55 // Apply all the extensions translators from a list
56 // Return true if one of the translators completed without raising an exception
57 // itself. Return of false indicates that if there are other translators
58 // available, they should be tried.
59 inline bool apply_exception_translators(std::forward_list<ExceptionTranslator>& translators) {
60  auto last_exception = std::current_exception();
61 
62  for (auto &translator : translators) {
63  try {
64  translator(last_exception);
65  return true;
66  } catch (...) {
67  last_exception = std::current_exception();
68  }
69  }
70  return false;
71 }

162  /* Store the function including any extra state it might have (e.g. a lambda capture object) */
163  // The unique_ptr makes sure nothing is leaked in case of an exception.
164  auto unique_rec = make_function_record();
165  auto rec = unique_rec.get();
166 
167  /* Store the capture object directly in the function record if there is enough space */
168  if (PYBIND11_SILENCE_MSVC_C4127(sizeof(capture) <= sizeof(rec->data))) {
169  /* Without these pragmas, GCC warns that there might not be
170  enough space to use the placement new operator. However, the
171  'if' statement above ensures that this is the case. */
172 #if defined(__GNUG__) && __GNUC__ >= 6 && !defined(__clang__) && !defined(__INTEL_COMPILER)
173 # pragma GCC diagnostic push
174 # pragma GCC diagnostic ignored "-Wplacement-new"
175 #endif
176  new ((capture *) &rec->data) capture { std::forward<Func>(f) };
177 #if defined(__GNUG__) && __GNUC__ >= 6 && !defined(__clang__) && !defined(__INTEL_COMPILER)
178 # pragma GCC diagnostic pop
179 #endif
180 #if defined(__GNUG__) && !PYBIND11_HAS_STD_LAUNDER && !defined(__INTEL_COMPILER)
181 # pragma GCC diagnostic push
182 # pragma GCC diagnostic ignored "-Wstrict-aliasing"
183 #endif
184  // UB without std::launder, but without breaking ABI and/or
185  // a significant refactoring it's "impossible" to solve.
186  if (!std::is_trivially_destructible<capture>::value)
187  rec->free_data = [](function_record *r) {
188  auto data = PYBIND11_STD_LAUNDER((capture *) &r->data);
189  (void) data;
190  data->~capture();
191  };
192 #if defined(__GNUG__) && !PYBIND11_HAS_STD_LAUNDER && !defined(__INTEL_COMPILER)
193 # pragma GCC diagnostic pop
194 #endif
195  } else {
196  rec->data[0] = new capture { std::forward<Func>(f) };
197  rec->free_data = [](function_record *r) { delete ((capture *) r->data[0]); };
198  }

200  /* Type casters for the function arguments and return value */
201  using cast_in = argument_loader<Args...>;
202  using cast_out = make_caster<
203  conditional_t<std::is_void<Return>::value, void_type, Return>
204  >;
205 
206  static_assert(expected_num_args<Extra...>(sizeof...(Args), cast_in::args_pos >= 0, cast_in::has_kwargs),
207  "The number of argument annotations does not match the number of function arguments");
208 
209  /* Dispatch code which converts function arguments and performs the actual function call */
210  rec->impl = [](function_call &call) -> handle {
211  cast_in args_converter;
212 
213  /* Try to cast the function arguments into the C++ domain */
214  if (!args_converter.load_args(call))
215  return PYBIND11_TRY_NEXT_OVERLOAD;
216 
217  /* Invoke call policy pre-call hook */
218  process_attributes<Extra...>::precall(call);
219 
220  /* Get a pointer to the capture object */
221  auto data = (sizeof(capture) <= sizeof(call.func.data)
222  ? &call.func.data : call.func.data[0]);
223  auto *cap = const_cast<capture *>(reinterpret_cast<const capture *>(data));
224 
225  /* Override policy for rvalues -- usually to enforce rvp::move on an rvalue */
226  return_value_policy policy = return_value_policy_override<Return>::policy(call.func.policy);
227 
228  /* Function scope guard -- defaults to the compile-to-nothing `void_type` */
229  using Guard = extract_guard_t<Extra...>;
230 
231  /* Perform the function call */
232  handle result = cast_out::cast(
233  std::move(args_converter).template call<Return, Guard>(cap->f), policy, call.parent);
234 
235  /* Invoke call policy post-call hook */
236  process_attributes<Extra...>::postcall(call, result);
237 
238  return result;
239  };

242  ? static_cast<std::uint16_t>(cast_in::args_pos)
243  : sizeof...(Args) - cast_in::has_kwargs; // Will get reduced more if we have a kw_only
244  rec->has_args = cast_in::args_pos >= 0;
245  rec->has_kwargs = cast_in::has_kwargs;

247  /* Process any user-provided function attributes */

304  void initialize_generic(unique_function_record &&unique_rec, const char *text,
305  const std::type_info *const *types, size_t args) {
306  // Do NOT receive `unique_rec` by value. If this function fails to move out the unique_ptr,
307  // we do not want this to destuct the pointer. `initialize` (the caller) still relies on the
308  // pointee being alive after this call. Only move out if a `capsule` is going to keep it alive.
309  auto rec = unique_rec.get();
310 
311  // Keep track of strdup'ed strings, and clean them up as long as the function's capsule
312  // has not taken ownership yet (when `unique_rec.release()` is called).
313  // Note: This cannot easily be fixed by a `unique_ptr` with custom deleter, because the strings
314  // are only referenced before strdup'ing. So only *after* the following block could `destruct`
315  // safely be called, but even then, `repr` could still throw in the middle of copying all strings.
316  strdup_guard guarded_strdup;
317 
318  /* Create copies of all referenced C-style strings */
319  rec->name = guarded_strdup(rec->name ? rec->name : "");
320  if (rec->doc) rec->doc = guarded_strdup(rec->doc);
321  for (auto &a: rec->args) {
322  if (a.name)
323  a.name = guarded_strdup(a.name);
324  if (a.descr)
325  a.descr = guarded_strdup(a.descr);
326  else if (a.value)
327  a.descr = guarded_strdup(repr(a.value).cast<std::string>().c_str());
328  }
329 
330  rec->is_constructor
331  = (strcmp(rec->name, "__init__") == 0) || (strcmp(rec->name, "__setstate__") == 0);
332 
333 #if !defined(NDEBUG) && !defined(PYBIND11_DISABLE_NEW_STYLE_INIT_WARNING)
334  if (rec->is_constructor && !rec->is_new_style_constructor) {
335  const auto class_name = detail::get_fully_qualified_tp_name((PyTypeObject *) rec->scope.ptr());
336  const auto func_name = std::string(rec->name);
337  PyErr_WarnEx(
338  PyExc_FutureWarning,
339  ("pybind11-bound class '" + class_name + "' is using an old-style "
340  "placement-new '" + func_name + "' which has been deprecated. See "
341  "the upgrade guide in pybind11's docs. This message is only visible "
342  "when compiled in debug mode.").c_str(), 0
343  );
344  }
345 #endif
346 
347  /* Generate a proper function signature */
348  std::string signature;
349  size_t type_index = 0, arg_index = 0;
350  bool is_starred = false;
351  for (auto *pc = text; *pc != '\0'; ++pc) {
352  const auto c = *pc;
353 
354  if (c == '{') {
355  // Write arg name for everything except *args and **kwargs.
356  is_starred = *(pc + 1) == '*';
357  if (is_starred)
358  continue;
359  // Separator for keyword-only arguments, placed before the kw
360  // arguments start (unless we are already putting an *args)
361  if (!rec->has_args && arg_index == rec->nargs_pos)
362  signature += "*, ";
363  if (arg_index < rec->args.size() && rec->args[arg_index].name) {
364  signature += rec->args[arg_index].name;
365  } else if (arg_index == 0 && rec->is_method) {
366  signature += "self";
367  } else {
368  signature += "arg" + std::to_string(arg_index - (rec->is_method ? 1 : 0));
369  }
370  signature += ": ";
371  } else if (c == '}') {
372  // Write default value if available.
373  if (!is_starred && arg_index < rec->args.size() && rec->args[arg_index].descr) {
374  signature += " = ";
375  signature += rec->args[arg_index].descr;
376  }
377  // Separator for positional-only arguments (placed after the
378  // argument, rather than before like *
379  if (rec->nargs_pos_only > 0 && (arg_index + 1) == rec->nargs_pos_only)
380  signature += ", /";
381  if (!is_starred)
382  arg_index++;
383  } else if (c == '%') {
384  const std::type_info *t = types[type_index++];
385  if (!t)
386  pybind11_fail("Internal error while parsing type signature (1)");
387  if (auto tinfo = detail::get_type_info(*t)) {
388  handle th((PyObject *) tinfo->type);
389  signature +=
390  th.attr("__module__").cast<std::string>() + "." +
391  th.attr("__qualname__").cast<std::string>(); // Python 3.3+, but we backport it to earlier versions
392  } else if (rec->is_new_style_constructor && arg_index == 0) {
393  // A new-style `__init__` takes `self` as `value_and_holder`.
394  // Rewrite it to the proper class type.
395  signature +=
396  rec->scope.attr("__module__").cast<std::string>() + "." +
397  rec->scope.attr("__qualname__").cast<std::string>();
398  } else {
399  std::string tname(t->name());
400  detail::clean_type_id(tname);
401  signature += tname;
402  }
403  } else {
404  signature += c;
405  }
406  }
407 
408  if (arg_index != args - rec->has_args - rec->has_kwargs || types[type_index] != nullptr)
409  pybind11_fail("Internal error while parsing type signature (2)");
410 
411 #if PY_MAJOR_VERSION < 3
412  if (strcmp(rec->name, "__next__") == 0) {
413  std::free(rec->name);
414  rec->name = guarded_strdup("next");
415  } else if (strcmp(rec->name, "__bool__") == 0) {
416  std::free(rec->name);
417  rec->name = guarded_strdup("__nonzero__");
418  }
419 #endif
420  rec->signature = guarded_strdup(signature.c_str());
421  rec->args.shrink_to_fit();
422  rec->nargs = (std::uint16_t) args;
423 
424  if (rec->sibling && PYBIND11_INSTANCE_METHOD_CHECK(rec->sibling.ptr()))
425  rec->sibling = PYBIND11_INSTANCE_METHOD_GET_FUNCTION(rec->sibling.ptr());
426 
427  detail::function_record *chain = nullptr, *chain_start = rec;
428  if (rec->sibling) {
429  if (PyCFunction_Check(rec->sibling.ptr())) {
430  auto *self = PyCFunction_GET_SELF(rec->sibling.ptr());
431  capsule rec_capsule = isinstance<capsule>(self) ? reinterpret_borrow<capsule>(self) : capsule(self);
432  chain = (detail::function_record *) rec_capsule;
433  /* Never append a method to an overload chain of a parent class;
434  instead, hide the parent's overloads in this case */
435  if (!chain->scope.is(rec->scope))
436  chain = nullptr;
437  }
438  // Don't trigger for things like the default __init__, which are wrapper_descriptors that we are intentionally replacing
439  else if (!rec->sibling.is_none() && rec->name[0] != '_')
440  pybind11_fail("Cannot overload existing non-function object \"" + std::string(rec->name) +
441  "\" with a function of the same name");
442  }
443 
444  if (!chain) {
445  /* No existing overload was found, create a new function object */
446  rec->def = new PyMethodDef();
447  std::memset(rec->def, 0, sizeof(PyMethodDef));
448  rec->def->ml_name = rec->name;
449  rec->def->ml_meth
450  = reinterpret_cast<PyCFunction>(reinterpret_cast<void (*)()>(dispatcher));
451  rec->def->ml_flags = METH_VARARGS | METH_KEYWORDS;
452 
453  capsule rec_capsule(unique_rec.release(), [](void *ptr) {
454  destruct((detail::function_record *) ptr);
455  });
456  guarded_strdup.release();
457 
458  object scope_module;
459  if (rec->scope) {
460  if (hasattr(rec->scope, "__module__")) {
461  scope_module = rec->scope.attr("__module__");
462  } else if (hasattr(rec->scope, "__name__")) {
463  scope_module = rec->scope.attr("__name__");
464  }
465  }
466 
467  m_ptr = PyCFunction_NewEx(rec->def, rec_capsule.ptr(), scope_module.ptr());
468  if (!m_ptr)
469  pybind11_fail("cpp_function::cpp_function(): Could not allocate function object");
470  } else {
471  /* Append at the beginning or end of the overload chain */
472  m_ptr = rec->sibling.ptr();
473  inc_ref();
474  if (chain->is_method != rec->is_method)
475  pybind11_fail("overloading a method with both static and instance methods is not supported; "
476  #if defined(NDEBUG)
477  "compile in debug mode for more details"
478  #else
479  "error while attempting to bind " + std::string(rec->is_method ? "instance" : "static") + " method " +
480  std::string(pybind11::str(rec->scope.attr("__name__"))) + "." + std::string(rec->name) + signature
481  #endif
482  );
483 
484  if (rec->prepend) {
485  // Beginning of chain; we need to replace the capsule's current head-of-the-chain
486  // pointer with this one, then make this one point to the previous head of the
487  // chain.
488  chain_start = rec;
489  rec->next = chain;
490  auto rec_capsule = reinterpret_borrow<capsule>(((PyCFunctionObject *) m_ptr)->m_self);
491  rec_capsule.set_pointer(unique_rec.release());
492  guarded_strdup.release();
493  } else {
494  // Or end of chain (normal behavior)
495  chain_start = chain;
496  while (chain->next)
497  chain = chain->next;
498  chain->next = unique_rec.release();
499  guarded_strdup.release();
500  }
501  }
502 
503  std::string signatures;
504  int index = 0;
505  /* Create a nice pydoc rec including all signatures and
506  docstrings of the functions in the overload chain */
507  if (chain && options::show_function_signatures()) {
508  // First a generic signature
509  signatures += rec->name;
510  signatures += "(*args, **kwargs)\n";
511  signatures += "Overloaded function.\n\n";
512  }
513  // Then specific overload signatures
514  bool first_user_def = true;
515  for (auto it = chain_start; it != nullptr; it = it->next) {
516  if (options::show_function_signatures()) {
517  if (index > 0) signatures += "\n";
518  if (chain)
519  signatures += std::to_string(++index) + ". ";
520  signatures += rec->name;
521  signatures += it->signature;
522  signatures += "\n";
523  }
524  if (it->doc && it->doc[0] != '\0' && options::show_user_defined_docstrings()) {
525  // If we're appending another docstring, and aren't printing function signatures, we
526  // need to append a newline first:
527  if (!options::show_function_signatures()) {
528  if (first_user_def) first_user_def = false;
529  else signatures += "\n";
530  }
531  if (options::show_function_signatures()) signatures += "\n";
532  signatures += it->doc;
533  if (options::show_function_signatures()) signatures += "\n";
534  }
535  }
536 
537  /* Install docstring */
538  auto *func = (PyCFunctionObject *) m_ptr;
539  std::free(const_cast<char *>(func->m_ml->ml_doc));
540  // Install docstring if it's non-empty (when at least one option is enabled)
541  func->m_ml->ml_doc
542  = signatures.empty() ? nullptr : PYBIND11_COMPAT_STRDUP(signatures.c_str());
543 
544  if (rec->is_method) {
545  m_ptr = PYBIND11_INSTANCE_METHOD_NEW(m_ptr, rec->scope.ptr());
546  if (!m_ptr)
547  pybind11_fail("cpp_function::cpp_function(): Could not allocate instance method object");
548  Py_DECREF(func);
549  }
550  }

553  static void destruct(detail::function_record *rec, bool free_strings = true) {
554  // If on Python 3.9, check the interpreter "MICRO" (patch) version.
555  // If this is running on 3.9.0, we have to work around a bug.
556  #if !defined(PYPY_VERSION) && PY_MAJOR_VERSION == 3 && PY_MINOR_VERSION == 9
557  static bool is_zero = Py_GetVersion()[4] == '0';
558  #endif
559 
560  while (rec) {
561  detail::function_record *next = rec->next;
562  if (rec->free_data)
563  rec->free_data(rec);
564  // During initialization, these strings might not have been copied yet,
565  // so they cannot be freed. Once the function has been created, they can.
566  // Check `make_function_record` for more details.
567  if (free_strings) {
568  std::free((char *) rec->name);
569  std::free((char *) rec->doc);
570  std::free((char *) rec->signature);
571  for (auto &arg: rec->args) {
572  std::free(const_cast<char *>(arg.name));
573  std::free(const_cast<char *>(arg.descr));
574  }
575  }
576  for (auto &arg: rec->args)
577  arg.value.dec_ref();
578  if (rec->def) {
579  std::free(const_cast<char *>(rec->def->ml_doc));
580  // Python 3.9.0 decref's these in the wrong order; rec->def
581  // If loaded on 3.9.0, let these leak (use Python 3.9.1 at runtime to fix)
582  // See https://github.com/python/cpython/pull/22670
583  #if !defined(PYPY_VERSION) && PY_MAJOR_VERSION == 3 && PY_MINOR_VERSION == 9
584  if (!is_zero)
585  delete rec->def;
586  #else
587  delete rec->def;
588  #endif
589  }
590  delete rec;
591  rec = next;
592  }

597  static PyObject *dispatcher(PyObject *self, PyObject *args_in, PyObject *kwargs_in) {
598  using namespace detail;
599 
600  /* Iterator over the list of potentially admissible overloads */
601  const function_record *overloads = (function_record *) PyCapsule_GetPointer(self, nullptr),
602  *it = overloads;
603 
604  /* Need to know how many arguments + keyword arguments there are to pick the right overload */
605  const auto n_args_in = (size_t) PyTuple_GET_SIZE(args_in);
606 
607  handle parent = n_args_in > 0 ? PyTuple_GET_ITEM(args_in, 0) : nullptr,
608  result = PYBIND11_TRY_NEXT_OVERLOAD;
609 
610  auto self_value_and_holder = value_and_holder();
611  if (overloads->is_constructor) {
612  if (!parent || !PyObject_TypeCheck(parent.ptr(), (PyTypeObject *) overloads->scope.ptr())) {
613  PyErr_SetString(PyExc_TypeError, "__init__(self, ...) called with invalid or missing `self` argument");
614  return nullptr;
615  }
616 
617  const auto tinfo = get_type_info((PyTypeObject *) overloads->scope.ptr());
618  const auto pi = reinterpret_cast<instance *>(parent.ptr());
619  self_value_and_holder = pi->get_value_and_holder(tinfo, true);
620 
621  // If this value is already registered it must mean __init__ is invoked multiple times;
622  // we really can't support that in C++, so just ignore the second __init__.
623  if (self_value_and_holder.instance_registered())
624  return none().release().ptr();
625  }
626 
627  try {
628  // We do this in two passes: in the first pass, we load arguments with `convert=false`;
629  // in the second, we allow conversion (except for arguments with an explicit
630  // py::arg().noconvert()). This lets us prefer calls without conversion, with
631  // conversion as a fallback.
632  std::vector<function_call> second_pass;
633 
634  // However, if there are no overloads, we can just skip the no-convert pass entirely
635  const bool overloaded = it != nullptr && it->next != nullptr;
636 
637  for (; it != nullptr; it = it->next) {
638 
639  /* For each overload:
640  1. Copy all positional arguments we were given, also checking to make sure that
641  named positional arguments weren't *also* specified via kwarg.
642  2. If we weren't given enough, try to make up the omitted ones by checking
643  whether they were provided by a kwarg matching the `py::arg("name")` name. If
644  so, use it (and remove it from kwargs); if not, see if the function binding
645  provided a default that we can use.
646  3. Ensure that either all keyword arguments were "consumed", or that the function
647  takes a kwargs argument to accept unconsumed kwargs.
648  4. Any positional arguments still left get put into a tuple (for args), and any
649  leftover kwargs get put into a dict.
650  5. Pack everything into a vector; if we have py::args or py::kwargs, they are an
651  extra tuple or dict at the end of the positional arguments.
652  6. Call the function call dispatcher (function_record::impl)
653 
654  If one of these fail, move on to the next overload and keep trying until we get a
655  result other than PYBIND11_TRY_NEXT_OVERLOAD.
656  */
657 
658  const function_record &func = *it;
659  size_t num_args = func.nargs; // Number of positional arguments that we need
660  if (func.has_args) --num_args; // (but don't count py::args
661  if (func.has_kwargs) --num_args; // or py::kwargs)
662  size_t pos_args = func.nargs_pos;
663 
664  if (!func.has_args && n_args_in > pos_args)
665  continue; // Too many positional arguments for this overload
666 
667  if (n_args_in < pos_args && func.args.size() < pos_args)
668  continue; // Not enough positional arguments given, and not enough defaults to fill in the blanks
669 
670  function_call call(func, parent);
671 
672  size_t args_to_copy = (std::min)(pos_args, n_args_in); // Protect std::min with parentheses
673  size_t args_copied = 0;
674 
675  // 0. Inject new-style `self` argument
676  if (func.is_new_style_constructor) {
677  // The `value` may have been preallocated by an old-style `__init__`
678  // if it was a preceding candidate for overload resolution.
679  if (self_value_and_holder)
680  self_value_and_holder.type->dealloc(self_value_and_holder);
681 
682  call.init_self = PyTuple_GET_ITEM(args_in, 0);
683  call.args.emplace_back(reinterpret_cast<PyObject *>(&self_value_and_holder));
684  call.args_convert.push_back(false);
685  ++args_copied;
686  }
687 
688  // 1. Copy any position arguments given.
689  bool bad_arg = false;
690  for (; args_copied < args_to_copy; ++args_copied) {
691  const argument_record *arg_rec = args_copied < func.args.size() ? &func.args[args_copied] : nullptr;
692  if (kwargs_in && arg_rec && arg_rec->name && dict_getitemstring(kwargs_in, arg_rec->name)) {
693  bad_arg = true;
694  break;
695  }
696 
697  handle arg(PyTuple_GET_ITEM(args_in, args_copied));
698  if (arg_rec && !arg_rec->none && arg.is_none()) {
699  bad_arg = true;
700  break;
701  }
702  call.args.push_back(arg);
703  call.args_convert.push_back(arg_rec ? arg_rec->convert : true);
704  }
705  if (bad_arg)
706  continue; // Maybe it was meant for another overload (issue #688)
707 
708  // Keep track of how many position args we copied out in case we need to come back
709  // to copy the rest into a py::args argument.
710  size_t positional_args_copied = args_copied;
711 
712  // We'll need to copy this if we steal some kwargs for defaults
713  dict kwargs = reinterpret_borrow<dict>(kwargs_in);
714 
715  // 1.5. Fill in any missing pos_only args from defaults if they exist
716  if (args_copied < func.nargs_pos_only) {
717  for (; args_copied < func.nargs_pos_only; ++args_copied) {
718  const auto &arg_rec = func.args[args_copied];
719  handle value;
720 
721  if (arg_rec.value) {
722  value = arg_rec.value;
723  }
724  if (value) {
725  call.args.push_back(value);
726  call.args_convert.push_back(arg_rec.convert);
727  } else
728  break;
729  }
730 
731  if (args_copied < func.nargs_pos_only)
732  continue; // Not enough defaults to fill the positional arguments
733  }
734 
735  // 2. Check kwargs and, failing that, defaults that may help complete the list
736  if (args_copied < num_args) {
737  bool copied_kwargs = false;
738 
739  for (; args_copied < num_args; ++args_copied) {
740  const auto &arg_rec = func.args[args_copied];
741 
742  handle value;
743  if (kwargs_in && arg_rec.name)
744  value = dict_getitemstring(kwargs.ptr(), arg_rec.name);
745 
746  if (value) {
747  // Consume a kwargs value
748  if (!copied_kwargs) {
749  kwargs = reinterpret_steal<dict>(PyDict_Copy(kwargs.ptr()));
750  copied_kwargs = true;
751  }
752  if (PyDict_DelItemString(kwargs.ptr(), arg_rec.name) == -1) {
753  throw error_already_set();
754  }
755  } else if (arg_rec.value) {
756  value = arg_rec.value;
757  }
758 
759  if (!arg_rec.none && value.is_none()) {
760  break;
761  }
762 
763  if (value) {
764  // If we're at the py::args index then first insert a stub for it to be replaced later
765  if (func.has_args && call.args.size() == func.nargs_pos)
766  call.args.push_back(none());
767 
768  call.args.push_back(value);
769  call.args_convert.push_back(arg_rec.convert);
770  }
771  else
772  break;
773  }
774 
775  if (args_copied < num_args)
776  continue; // Not enough arguments, defaults, or kwargs to fill the positional arguments
777  }
778 
779  // 3. Check everything was consumed (unless we have a kwargs arg)
780  if (kwargs && !kwargs.empty() && !func.has_kwargs)
781  continue; // Unconsumed kwargs, but no py::kwargs argument to accept them
782 
783  // 4a. If we have a py::args argument, create a new tuple with leftovers
784  if (func.has_args) {
785  tuple extra_args;
786  if (args_to_copy == 0) {
787  // We didn't copy out any position arguments from the args_in tuple, so we
788  // can reuse it directly without copying:
789  extra_args = reinterpret_borrow<tuple>(args_in);
790  } else if (positional_args_copied >= n_args_in) {
791  extra_args = tuple(0);
792  } else {
793  size_t args_size = n_args_in - positional_args_copied;
794  extra_args = tuple(args_size);
795  for (size_t i = 0; i < args_size; ++i) {
796  extra_args[i] = PyTuple_GET_ITEM(args_in, positional_args_copied + i);
797  }
798  }
799  if (call.args.size() <= func.nargs_pos)
800  call.args.push_back(extra_args);
801  else
802  call.args[func.nargs_pos] = extra_args;
803  call.args_convert.push_back(false);
804  call.args_ref = std::move(extra_args);
805  }
806 
807  // 4b. If we have a py::kwargs, pass on any remaining kwargs
808  if (func.has_kwargs) {
809  if (!kwargs.ptr())
810  kwargs = dict(); // If we didn't get one, send an empty one
811  call.args.push_back(kwargs);
812  call.args_convert.push_back(false);
813  call.kwargs_ref = std::move(kwargs);
814  }
815 
816  // 5. Put everything in a vector. Not technically step 5, we've been building it
817  // in `call.args` all along.
818  #if !defined(NDEBUG)
819  if (call.args.size() != func.nargs || call.args_convert.size() != func.nargs)
820  pybind11_fail("Internal error: function call dispatcher inserted wrong number of arguments!");
821  #endif
822 
823  std::vector<bool> second_pass_convert;
824  if (overloaded) {
825  // We're in the first no-convert pass, so swap out the conversion flags for a
826  // set of all-false flags. If the call fails, we'll swap the flags back in for
827  // the conversion-allowed call below.
828  second_pass_convert.resize(func.nargs, false);
829  call.args_convert.swap(second_pass_convert);
830  }
831 
832  // 6. Call the function.
833  try {
834  loader_life_support guard{};
835  result = func.impl(call);
836  } catch (reference_cast_error &) {
837  result = PYBIND11_TRY_NEXT_OVERLOAD;
838  }
839 
840  if (result.ptr() != PYBIND11_TRY_NEXT_OVERLOAD)
841  break;
842 
843  if (overloaded) {
844  // The (overloaded) call failed; if the call has at least one argument that
845  // permits conversion (i.e. it hasn't been explicitly specified `.noconvert()`)
846  // then add this call to the list of second pass overloads to try.
847  for (size_t i = func.is_method ? 1 : 0; i < pos_args; i++) {
848  if (second_pass_convert[i]) {
849  // Found one: swap the converting flags back in and store the call for
850  // the second pass.
851  call.args_convert.swap(second_pass_convert);
852  second_pass.push_back(std::move(call));
853  break;
854  }
855  }
856  }
857  }
858 
859  if (overloaded && !second_pass.empty() && result.ptr() == PYBIND11_TRY_NEXT_OVERLOAD) {
860  // The no-conversion pass finished without success, try again with conversion allowed
861  for (auto &call : second_pass) {
862  try {
863  loader_life_support guard{};
864  result = call.func.impl(call);
865  } catch (reference_cast_error &) {
866  result = PYBIND11_TRY_NEXT_OVERLOAD;
867  }
868 
869  if (result.ptr() != PYBIND11_TRY_NEXT_OVERLOAD) {
870  // The error reporting logic below expects 'it' to be valid, as it would be
871  // if we'd encountered this failure in the first-pass loop.
872  if (!result)
873  it = &call.func;
874  break;
875  }
876  }
877  }
878  } catch (error_already_set &e) {
879  e.restore();
880  return nullptr;
881 #ifdef __GLIBCXX__
882  } catch ( abi::__forced_unwind& ) {
883  throw;
884 #endif
885  } catch (...) {
886  /* When an exception is caught, give each registered exception
887  translator a chance to translate it to a Python exception. First
888  all module-local translators will be tried in reverse order of
889  registration. If none of the module-locale translators handle
890  the exception (or there are no module-locale translators) then
891  the global translators will be tried, also in reverse order of
892  registration.
893 
894  A translator may choose to do one of the following:
895 
896  - catch the exception and call PyErr_SetString or PyErr_SetObject
897  to set a standard (or custom) Python exception, or
898  - do nothing and let the exception fall through to the next translator, or
899  - delegate translation to the next translator by throwing a new type of exception. */
900 
901  auto &local_exception_translators = get_local_internals().registered_exception_translators;
902  if (detail::apply_exception_translators(local_exception_translators)) {
903  return nullptr;
904  }
905  auto &exception_translators = get_internals().registered_exception_translators;
906  if (detail::apply_exception_translators(exception_translators)) {
907  return nullptr;
908  }
909 
910  PyErr_SetString(PyExc_SystemError, "Exception escaped from default exception translator!");
911  return nullptr;
912  }

915  if (msg.find("std::") != std::string::npos) {
916  msg += "\n\n"
917  "Did you forget to `#include <pybind11/stl.h>`? Or <pybind11/complex.h>,\n"

921  }

926  return handle(Py_NotImplemented).inc_ref().ptr();
927

929  std::string(overloads->is_constructor ? "constructor" : "function") +
930  " arguments. The following argument types are supported:\n";
931

934  msg += " "+ std::to_string(++ctr) + ". ";
935

939  std::string sig = it2->signature;
940  size_t start = sig.find('(') + 7; // skip "(self: "
941  if (start < sig.size()) {
942  // End at the , for the next argument
943  size_t end = sig.find(", "), next = end + 2;
944  size_t ret = sig.rfind(" -> ");
945  // Or the ), if there is no comma:
946  if (end >= sig.size()) next = end = sig.find(')');
947  if (start < end && next < sig.size()) {
948  msg.append(sig, start, end - start);
949  msg += '(';
950  msg.append(sig, next, ret - next);
951  wrote_sig = true;
952  }
953  }
954  }
955  if (!wrote_sig) msg += it2->signature;
956 
957  msg += "\n";
958  }
959  msg += "\nInvoked with: ";
960  auto args_ = reinterpret_borrow<tuple>(args_in);
961  bool some_args = false;
962  for (size_t ti = overloads->is_constructor ? 1 : 0; ti < args_.size(); ++ti) {
963  if (!some_args) some_args = true;
964  else msg += ", ";
965  try {
966  msg += pybind11::repr(args_[ti]);
967  } catch (const error_already_set&) {
968  msg += "<repr raised Error>";
969  }
970  }
971  if (kwargs_in) {
972  auto kwargs = reinterpret_borrow<dict>(kwargs_in);
973  if (!kwargs.empty()) {
974  if (some_args) msg += "; ";
975  msg += "kwargs: ";
976  bool first = true;
977  for (auto kwarg : kwargs) {
978  if (first) first = false;
979  else msg += ", ";
980  msg += pybind11::str("{}=").format(kwarg.first);
981  try {
982  msg += pybind11::repr(kwarg.second);
983  } catch (const error_already_set&) {
984  msg += "<repr raised Error>";
985  }
986  }
987  }
988  }
989 
990  append_note_if_missing_header_is_suspected(msg);
991  PyErr_SetString(PyExc_TypeError, msg.c_str());
992  return nullptr;
993  }
994  if (!result) {
995  std::string msg = "Unable to convert function return value to a "
996  "Python type! The signature was\n\t";
997  msg += it->signature;
998  append_note_if_missing_header_is_suspected(msg);
999  PyErr_SetString(PyExc_TypeError, msg.c_str());
1000  return nullptr;
1001  }
1002  if (overloads->is_constructor && !self_value_and_holder.holder_constructed()) {
1003  auto *pi = reinterpret_cast<instance *>(parent.ptr());
1004  self_value_and_holder.type->init_instance(pi, nullptr);
1005  }
1006  return result.ptr();
1007  }
