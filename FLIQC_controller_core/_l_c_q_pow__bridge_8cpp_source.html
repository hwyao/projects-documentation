247 void LCQPow_bridge::runSolver(const LCQProblemInput &input, LCQProblemOutput &output){
248 #ifdef FLIQC_DEBUG
249 // check the size of the input matches what is needed
250 #endif
251
252 this->pimpl->lcqp.reset(new LCQPow::LCQProblem(this->nVariables, this->nConstraints, this->nComplementarity));
253 this->pimpl->lcqp->setOptions(*(this->pimpl->options));
254
255 LCQPow::ReturnValue retVal = this->pimpl->lcqp->loadLCQP(
256 input.Q.data(), input.g.data(),
257 input.L.data(), input.R.data(), input.lbL.data(), input.ubL.data(), input.lbR.data(), input.ubR.data(),
258 input.A.data(), input.lbA.data(), input.ubA.data(),
259 input.lb.data(), input.ub.data(),
260 input.x0.data(), input.y0.data()
261 );
262 if (retVal != LCQPow::SUCCESSFUL_RETURN){
263 throw std::runtime_error("Failed to load LCQP problem.");
264 }
265
266 retVal = this->pimpl->lcqp->runSolver();
267 if (retVal != LCQPow::SUCCESSFUL_RETURN){
268 LCQProblemDebug debug = this->getDebugStatistics();
269 throw LCQPowException(input, output, debug);
270 }
271
272 double xOut[this->nVariables];
273 this->pimpl->lcqp->getPrimalSolution(xOut);
274 double yOut[this->nVariables + this->nConstraints + 2*this->nComplementarity];
275 this->pimpl->lcqp->getDualSolution(yOut);
276 output.x = Eigen::Map<Eigen::VectorXd>(xOut, this->nVariables);
277 output.y = Eigen::Map<Eigen::VectorXd>(yOut, this->nVariables + this->nConstraints + 2*this->nComplementarity);
278 }
