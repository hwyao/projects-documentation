233  void LCQPow_bridge::runSolver(const LCQProblemInput &input, LCQProblemOutput &output){
234  #ifdef FLIQC_DEBUG
235  // check the size of the input matches what is needed
236  #endif
237 
238  this->pimpl->lcqp.reset(new LCQPow::LCQProblem(this->nVariables, this->nConstraints, this->nComplementarity));
239  this->pimpl->lcqp->setOptions(*(this->pimpl->options));
240 
241  LCQPow::ReturnValue retVal = this->pimpl->lcqp->loadLCQP(
242  input.Q.data(), input.g.data(),
243  input.L.data(), input.R.data(), input.lbL.data(), input.ubL.data(), input.lbR.data(), input.ubR.data(),
244  input.A.data(), input.lbA.data(), input.ubA.data(),
245  input.lb.data(), input.ub.data(),
246  input.x0.data(), input.y0.data()
247  );
248  if (retVal != LCQPow::SUCCESSFUL_RETURN){
249  throw std::runtime_error("Failed to load LCQP problem.");
250  }
251 
252  retVal = this->pimpl->lcqp->runSolver();
253  if (retVal != LCQPow::SUCCESSFUL_RETURN){
254  LCQProblemDebug debug = this->getDebugStatistics();
255  throw LCQPowException(input, output, debug);
256  }
257 
258  double xOut[this->nVariables];
259  this->pimpl->lcqp->getPrimalSolution(xOut);
260  double yOut[this->nVariables + this->nConstraints + 2*this->nComplementarity];
261  this->pimpl->lcqp->getDualSolution(yOut);
262  output.x = Eigen::Map<Eigen::VectorXd>(xOut, this->nVariables);
263  output.y = Eigen::Map<Eigen::VectorXd>(yOut, this->nVariables + this->nConstraints + 2*this->nComplementarity);
264  }
