46  options.setPrintLevel(PrintLevel::INNER_LOOP_ITERATES);

53  double* Q = new double[nV*nV]();
54  double* g = new double[nV]();
55  double* L = new double[nComp*nV]();
56  double* R = new double[nComp*nV]();
57  double* A = new double[nC*nV]();
58  double* lbA = new double[nC]();

64  // Assign problem data
65  Q[0] = 17; Q[1*nV + 1] = 17;
66  Q[0*nV + 1] = -15;
67  Q[1*nV + 0] = -15;
68 
69  // Regularization on Q
70  for (int i = 2; i < nV; i++)
71  Q[i*nV + i] = 5e-12;
72 
73  // Objective linear term
74  double Qx[2*2] = {17, -15, -15, 17};
75  LCQPow::Utilities::MatrixMultiplication(Qx, x_ref, g, 2, 2, 1);
76  LCQPow::Utilities::WeightedVectorAdd(-1, g, 0, g, g, 2);
77 
78  // Constraints
79  for (int i = 0; i < N; i++) {
80  // Equality constraint ([cos sin]'*x + lambda = 1)
81  A[i*nV + 0] = cos((2*M_PI*i)/N);
82  A[i*nV + 1] = sin((2*M_PI*i)/N);
83  A[i*nV + 2 + 2*i] = 1;
84 
85  // Convex combination constraint (sum theta = 1)
86  A[N*nV + 3 + 2*i] = 1;
87 
88  // Complementarity constraints (lambda*theta = 0)
89  L[i*nV + 2 + 2*i] = 1;
90  R[i*nV + 3 + 2*i] = 1;
91 
92  // constraint bounds
93  lbA[i] = 1;
94  ubA[i] = 1;
95 
96  x0[2*i + 2] = 1;
97  x0[2*i + 3] = 1;
98  }
99 
100  // Constraint bound for last constraint
101  lbA[N] = 1;
102  ubA[N] = 1;
103 
104  // Load data
105  ReturnValue retVal = lcqp.loadLCQP( Q, g, L, R, 0, 0, 0, 0, A, lbA, ubA, 0, 0, x0 );
106 
107  if (retVal != SUCCESSFUL_RETURN)
108  {

111  }

143  xOpt[0],xOpt[1],
