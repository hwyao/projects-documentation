26#include <iostream>
27#include "LCQProblem.hpp"
28
29using namespace LCQPow;
30
31int main() {
32 std::cout << "Preparing unit circle optimization problem...\n";
33
34 // Set dimensions
35 int N = 100;
36 int nV = 2 + 2*N;
37 int nC = N+1;
38 int nComp = N;
39
40 // Reference
41 double x_ref[2] = {0.5, -0.6};
42
43 // Set up LCQP object
44 LCQProblem lcqp( nV, nC, nComp );
45 Options options;
46 options.setPrintLevel(PrintLevel::INNER_LOOP_ITERATES);
47 options.setQPSolver(QPSolver::OSQP_SPARSE);
48 options.setStationarityTolerance( 10e-3 );
49 lcqp.setOptions( options );
50
51
52 // Allocate vectors
53 double* Q = new double[nV*nV]();
54 double* g = new double[nV]();
55 double* L = new double[nComp*nV]();
56 double* R = new double[nComp*nV]();
57 double* A = new double[nC*nV]();
58 double* lbA = new double[nC]();
59 double* ubA = new double[nC]();
60 double* x0 = new double[nV]();
61 x0[0] = x_ref[0];
62 x0[1] = x_ref[1];
63
64 // Assign problem data
65 Q[0] = 17; Q[1*nV + 1] = 17;
66 Q[0*nV + 1] = -15;
67 Q[1*nV + 0] = -15;
68
69 // Regularization on Q
70 for (int i = 2; i < nV; i++)
71 Q[i*nV + i] = 5e-12;
72
73 // Objective linear term
74 double Qx[2*2] = {17, -15, -15, 17};
75 LCQPow::Utilities::MatrixMultiplication(Qx, x_ref, g, 2, 2, 1);
76 LCQPow::Utilities::WeightedVectorAdd(-1, g, 0, g, g, 2);
77
78 // Constraints
79 for (int i = 0; i < N; i++) {
80 // Equality constraint ([cos sin]'*x + lambda = 1)
81 A[i*nV + 0] = cos((2*M_PI*i)/N);
82 A[i*nV + 1] = sin((2*M_PI*i)/N);
83 A[i*nV + 2 + 2*i] = 1;
84
85 // Convex combination constraint (sum theta = 1)
86 A[N*nV + 3 + 2*i] = 1;
87
88 // Complementarity constraints (lambda*theta = 0)
89 L[i*nV + 2 + 2*i] = 1;
90 R[i*nV + 3 + 2*i] = 1;
91
92 // constraint bounds
93 lbA[i] = 1;
94 ubA[i] = 1;
95
96 x0[2*i + 2] = 1;
97 x0[2*i + 3] = 1;
98 }
99
100 // Constraint bound for last constraint
101 lbA[N] = 1;
102 ubA[N] = 1;
103
104 // Load data
105 ReturnValue retVal = lcqp.loadLCQP( Q, g, L, R, 0, 0, 0, 0, A, lbA, ubA, 0, 0, x0 );
106
107 if (retVal != SUCCESSFUL_RETURN)
108 {
109 printf("Failed to load LCQP.\n");
110 return 1;
111 }
112
113 // Must switch to sparse mode (if using a sparse solver)
114 if (options.getQPSolver() >= LCQPow::QPOASES_SPARSE) {
115 retVal = lcqp.switchToSparseMode( );
116
117 if (retVal != SUCCESSFUL_RETURN)
118 {
119 printf("Failed to switch to sparse mode LCQP.\n");
120 return 1;
121 }
122 }
123
124 // Solve the LCQP
125 retVal = lcqp.runSolver();
126
127 if (retVal != SUCCESSFUL_RETURN)
128 {
129 printf("Failed to solve LCQP.\n");
130 return 1;
131 }
132
133 // Get solutions
134 double* xOpt = new double[nV];
135 double* yOpt = new double[nV + nC + 2*nComp];
136 LCQPow::OutputStatistics stats;
137 lcqp.getPrimalSolution( xOpt );
138 lcqp.getDualSolution( yOpt );
139 lcqp.getOutputStatistics( stats );
140
141 // Print coordinates of solution and output statistics
142 printf( "\nxOpt = [ %g, %g ]; i = %d; k = %d; rho = %g; WSR = %d \n\n",
143 xOpt[0],xOpt[1],
144 stats.getIterTotal(), stats.getIterOuter(), stats.getRhoOpt(), stats.getSubproblemIter() );
145
146 // Print a reference to the global and local solutions
147 printf("For reference: Global solution is at: [ %g, %g ]\n", 0.1811, -0.9835);
148 printf(" Another local solution: [ %g, %g ]\n", 0.9764, -0.2183);
149
150 // Clean Up
151 delete[] xOpt; delete[] yOpt;
152 delete[] Q; delete[] g; delete[] L; delete[] R; delete[] A;
153 delete[] lbA; delete[] ubA; delete[] x0;
154
155 return 0;
156}
