178  void writeVariableAsCSV(const std::string& base_path, const T& variable, const std::string& variable_name) {
179  std::cout << "Starting to write variable: " << variable_name << " to CSV." << std::endl;
180 
181  std::ofstream file(base_path + "/" + variable_name + ".csv");
182  if (!file.is_open()) {
183  throw std::runtime_error("Failed to open file: " + base_path + "/" + variable_name + ".csv");
184  }
185 
186  if constexpr (std::is_arithmetic_v<T>) {
187  file << variable;
188  } else if constexpr (std::is_base_of_v<Eigen::MatrixBase<T>, T>) {
189  for (int i = 0; i < variable.rows(); ++i) {
190  for (int j = 0; j < variable.cols(); ++j) {
191  file << variable(i, j);
192  if (j < variable.cols() - 1) {
193  file << ",";
194  }
195  }
196  file << "\n";
197  }
198  } else if constexpr (std::is_same_v<T, std::vector<double>>) {
199  for (const auto& val : variable) {
200  file << val << "\n";
201  }
202  } else if constexpr (std::is_same_v<T, std::vector<int>>) {
203  for (const auto& val : variable) {
204  file << val << "\n";
205  }
206  } else if constexpr (std::is_same_v<T, std::vector<std::vector<double>>>) {
207  for (const auto& row : variable) {
208  for (size_t i = 0; i < row.size(); ++i) {
209  file << row[i];
210  if (i < row.size() - 1) {
211  file << ",";
212  }
213  }
214  file << "\n";
215  }
216  } else {
217  throw std::runtime_error("Unsupported variable type for CSV export: " + variable_name + " as " + typeid(T).name());
218  }
219 
220  file.close();
221  }
