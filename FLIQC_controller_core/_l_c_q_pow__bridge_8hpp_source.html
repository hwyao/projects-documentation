192 void writeVariableAsCSV(const std::string& base_path, const T& variable, const std::string& variable_name) {
193 std::cout << "Starting to write variable: " << variable_name << " to CSV." << std::endl;
194
195 std::ofstream file(base_path + "/" + variable_name + ".csv");
196 if (!file.is_open()) {
197 throw std::runtime_error("Failed to open file: " + base_path + "/" + variable_name + ".csv");
198 }
199
200 if constexpr (std::is_arithmetic_v<T>) {
201 file << variable;
202 } else if constexpr (std::is_base_of_v<Eigen::MatrixBase<T>, T>) {
203 for (int i = 0; i < variable.rows(); ++i) {
204 for (int j = 0; j < variable.cols(); ++j) {
205 file << variable(i, j);
206 if (j < variable.cols() - 1) {
207 file << ",";
208 }
209 }
210 file << "\n";
211 }
212 } else if constexpr (std::is_same_v<T, std::vector<double>>) {
213 for (const auto& val : variable) {
214 file << val << "\n";
215 }
216 } else if constexpr (std::is_same_v<T, std::vector<int>>) {
217 for (const auto& val : variable) {
218 file << val << "\n";
219 }
220 } else if constexpr (std::is_same_v<T, std::vector<std::vector<double>>>) {
221 for (const auto& row : variable) {
222 for (size_t i = 0; i < row.size(); ++i) {
223 file << row[i];
224 if (i < row.size() - 1) {
225 file << ",";
226 }
227 }
228 file << "\n";
229 }
230 } else {
231 throw std::runtime_error("Unsupported variable type for CSV export: " + variable_name + " as " + typeid(T).name());
232 }
233
234 file.close();
235 }
