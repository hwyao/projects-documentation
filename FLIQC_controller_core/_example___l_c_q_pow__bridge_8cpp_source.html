49 // Allocate vectors
50 input.Q = Eigen::MatrixXd::Zero(nV, nV);
51 input.g = Eigen::VectorXd::Zero(nV);
52 input.L = Eigen::MatrixXd::Zero(nComp, nV);
53 input.R = Eigen::MatrixXd::Zero(nComp, nV);
54 input.A = Eigen::MatrixXd::Zero(nC, nV);
55 input.lbA = Eigen::VectorXd::Zero(nC);
56 input.ubA = Eigen::VectorXd::Zero(nC);
57 input.x0 = Eigen::VectorXd::Zero(nV);
58
59 // Assign problem data
60 input.x0(0) = x_ref(0);
61 input.x0(1) = x_ref(1);
62 input.Q(0, 0) = 17;
63 input.Q(1, 1) = 17;
64 input.Q(0, 1) = -15;
65 input.Q(1, 0) = -15;
66
67 // Regularization on Q
68 for (int i = 2; i < nV; i++)
69 input.Q(i, i) = 5e-12;
70
71 // Objective linear term
72 Eigen::Matrix2d Qx;
73 Qx << 17, -15, -15, 17;
74 input.g.head(2) = Qx*x_ref;
75 input.g = -input.g;
76
77 // Constraints
78 for (int i = 0; i < N; i++) {
79 // Equality constraint ([cos sin]'*x + lambda = 1)
80 input.A(i, 0) = cos((2*M_PI*i)/N);
81 input.A(i, 1) = sin((2*M_PI*i)/N);
82 input.A(i, 2 + 2*i) = 1;
83
84 // Convex combination constraint (sum theta = 1)
85 input.A(N, 3 + 2*i) = 1;
86
87 // Complementarity constraints (lambda*theta = 0)
88 input.L(i, 2 + 2*i) = 1;
89 input.R(i, 3 + 2*i) = 1;
90
91 // constraint bounds
92 input.lbA(i) = 1;
93 input.ubA(i) = 1;
94
95 input.x0(2*i + 2) = 1;
96 input.x0(2*i + 3) = 1;
97 }
98
99 // Constraint bound for last constraint
100 input.lbA(N) = 1;
101 input.ubA(N) = 1;
102
103 runLQCPow(input, output);
