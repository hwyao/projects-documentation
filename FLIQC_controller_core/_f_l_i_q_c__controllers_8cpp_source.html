18  Eigen::VectorXd FLIQC_controller_joint_velocity_basic::runController(const FLIQC_state_input& state_input,
19  const std::vector<FLIQC_distance_input> &dist_inputs,
20  FLIQC_control_output& control_output) {
21  // filter out the active distance inputs: if the distance is less than the active threshold, then it is active
22  std::vector<FLIQC_distance_input> active_dist_inputs;
23  for (auto dist_input_i : dist_inputs) {
24  if (dist_input_i.distance <= active_threshold) {
25  active_dist_inputs.push_back(dist_input_i);
26  }
27  }
28 
29  if (active_dist_inputs.size() == 0) {
30  // no active distance inputs, return the guide velocity
31  return state_input.q_dot_guide;
32  }
33  int nContacts = active_dist_inputs.size();
34  int nVariables = nJoint + nContacts;
35 
36  // construct the input of the LCQProblem: [0] The cost input
37  Eigen::MatrixXd quad_cost = Eigen::MatrixXd::Zero(nJoint, nJoint);
38  Eigen::VectorXd lin_cost = Eigen::VectorXd::Zero(nJoint);
39  if (quad_cost_type == FLIQC_QUAD_COST_IDENTITY) {
40  quad_cost += Eigen::MatrixXd::Identity(nJoint, nJoint);
41  } else if (quad_cost_type == FLIQC_QUAD_COST_JOINT_VELOCITY_ERROR){
42  quad_cost += Eigen::MatrixXd::Identity(nJoint, nJoint);
43  lin_cost += -2 * state_input.q_dot_guide;
44  } else if (quad_cost_type == FLIQC_QUAD_COST_MASS_MATRIX) {
45  quad_cost += state_input.M;
46  } else if (quad_cost_type == FLIQC_QUAD_COST_MASS_MATRIX_VELOCITY_ERROR) {
47  Eigen::MatrixXd W = weight_on_mass_matrix.asDiagonal();
48  quad_cost += W * state_input.M;
49  lin_cost += -2 * W * state_input.M * state_input.q_dot_guide;
50  } else {
51  throw std::invalid_argument("The type of the cost function" + std::to_string(quad_cost_type) + " is not supported.");
52  }
53 
54  if (linear_cost_type == FLIQC_LINEAR_COST_NONE) {
55  //lin_cost += Eigen::VectorXd::Zero(nJoint); // do nothing
56  } else {
57  throw std::invalid_argument("The type of the linear cost function " + std::to_string(linear_cost_type) + " is not supported.");
58  }
59 
60  lcqp_input.Q = Eigen::MatrixXd::Zero(nVariables, nVariables);
61  lcqp_input.Q << quad_cost, Eigen::MatrixXd::Zero(nJoint, nContacts),
62  Eigen::MatrixXd::Zero(nContacts, nJoint), Eigen::MatrixXd::Identity(nContacts, nContacts) * lambda_cost_penalty;
63 
64  lcqp_input.g = Eigen::VectorXd::Zero(nVariables);
65  lcqp_input.g << lin_cost, Eigen::VectorXd::Zero(nContacts);
66 
67  // construct the input of the LCQProblem: [1] all the close-constant input.
68  lcqp_input.L = Eigen::MatrixXd::Zero(nContacts, nVariables);
69  lcqp_input.L.block(0, nJoint, nContacts, nContacts) = Eigen::MatrixXd::Identity(nContacts, nContacts);
70  lcqp_input.lbL = Eigen::VectorXd::Zero(nContacts);
71  if (enable_lambda_constraint_in_L){
72  lcqp_input.ubL = Eigen::VectorXd::Constant(nContacts, lambda_max);
73  }
74 
75  lcqp_input.R = Eigen::MatrixXd::Zero(nContacts, nVariables);
76  lcqp_input.lbR = Eigen::VectorXd::Zero(nContacts);
77  if (enable_esc_vel_constraint){
78  lcqp_input.ubR = Eigen::VectorXd::Constant(nContacts, esc_vel_max);
79  }
80 
81  lcqp_input.A = Eigen::MatrixXd::Zero(nJoint, nVariables);
82  lcqp_input.A.block(0, 0, nJoint, nJoint) = Eigen::MatrixXd::Identity(nJoint, nJoint);
83  lcqp_input.lbA = state_input.q_dot_guide;
84  lcqp_input.ubA = state_input.q_dot_guide;
85 
86  lcqp_input.lb = Eigen::VectorXd::Zero(nVariables);
87  lcqp_input.ub = Eigen::VectorXd::Constant(nVariables, std::numeric_limits<double>::max());
88  if (enable_lambda_constraint_in_x){
89  lcqp_input.ub.tail(nContacts) = Eigen::VectorXd::Constant(nContacts, lambda_max);
90  }
91 
92  if (q_dot_max.size() != 0) {
93  lcqp_input.lb.head(nJoint) = -q_dot_max;
94  lcqp_input.ub.head(nJoint) = q_dot_max;
95  }
96  else{
97  lcqp_input.lb.head(nJoint) = Eigen::VectorXd::Constant(nJoint, -std::numeric_limits<double>::max());
98  //lcqp_input.ub.head(nJoint) = Eigen::VectorXd::Constant(nJoint, std::numeric_limits<double>::max()); // already set
99  }
100 
101  // construct the input of the LCQProblem: [2] all the distance-dependent input.
102  Eigen::MatrixXd nullspace_projector;
103  if (enable_nullspace_projector_in_A) {
104  Eigen::MatrixXd Jpos = state_input.J.block(0, 0, 3, nJoint);
105  nullspace_projector = Eigen::MatrixXd::Identity(nJoint, nJoint) - Jpos.jacobiSvd(Eigen::ComputeThinU | Eigen::ComputeThinV).solve(Jpos);
106  } else {
107  nullspace_projector = Eigen::MatrixXd::Identity(nJoint, nJoint);
108  }
109  for (int i = 0; i < active_dist_inputs.size(); i++) {
110  // lbR
111  lcqp_input.lbR(i) = -active_dist_inputs[i].distance + eps;
112  // R
113  lcqp_input.R.block(i, 0, 1, nJoint) = dt * active_dist_inputs[i].projector_control_to_dist;
114  // A
115  if(active_dist_inputs[i].projector_dist_to_control.size() != 0){
116  lcqp_input.A.block(0, nJoint + i, nJoint, 1) = - nullspace_projector
117  * active_dist_inputs[i].projector_dist_to_control;
118  }
119  else{
120  // This is then the pseudo-inverse of the projector_control_to_dist vector
121  lcqp_input.A.block(0, nJoint + i, nJoint, 1) = - nullspace_projector
122  * (1.0 / active_dist_inputs[i].projector_control_to_dist.norm() / active_dist_inputs[i].projector_control_to_dist.norm())
123  * active_dist_inputs[i].projector_control_to_dist.transpose();
124  }
125  }
126 
127  // construct the input of the LCQProblem: [3] initial guess
128  lcqp_input.x0.resize(0);
129  lcqp_input.y0.resize(0);
130 
131  lcqp_solver.nVariables = nVariables;
132  lcqp_solver.nConstraints = nContacts;
133  lcqp_solver.nComplementarity = nContacts;
134 
135  lcqp_solver.runSolver(lcqp_input, lcqp_output);
136  control_output.x = lcqp_output.x;
137  control_output.y = lcqp_output.y;
138 
139  Eigen::VectorXd result = lcqp_output.x.head(nJoint);
140  return result;
141  }
