103 // Populates a just-created cache entry.
104 PYBIND11_NOINLINE void all_type_info_populate(PyTypeObject *t, std::vector<type_info *> &bases) {
105  std::vector<PyTypeObject *> check;
106  for (handle parent : reinterpret_borrow<tuple>(t->tp_bases))
107  check.push_back((PyTypeObject *) parent.ptr());
108 
109  auto const &type_dict = get_internals().registered_types_py;
110  for (size_t i = 0; i < check.size(); i++) {
111  auto type = check[i];
112  // Ignore Python2 old-style class super type:
113  if (!PyType_Check((PyObject *) type)) continue;
114 
115  // Check `type` in the current set of registered python types:
116  auto it = type_dict.find(type);
117  if (it != type_dict.end()) {
118  // We found a cache entry for it, so it's either pybind-registered or has pre-computed
119  // pybind bases, but we have to make sure we haven't already seen the type(s) before: we
120  // want to follow Python/virtual C++ rules that there should only be one instance of a
121  // common base.
122  for (auto *tinfo : it->second) {
123  // NB: Could use a second set here, rather than doing a linear search, but since
124  // having a large number of immediate pybind11-registered types seems fairly
125  // unlikely, that probably isn't worthwhile.
126  bool found = false;
127  for (auto *known : bases) {
128  if (known == tinfo) { found = true; break; }
129  }
130  if (!found) bases.push_back(tinfo);
131  }
132  }
133  else if (type->tp_bases) {
134  // It's some python type, so keep follow its bases classes to look for one or more
135  // registered types
136  if (i + 1 == check.size()) {
137  // When we're at the end, we can pop off the current element to avoid growing
138  // `check` when adding just one base (which is typical--i.e. when there is no
139  // multiple inheritance)
140  check.pop_back();
141  i--;
142  }
143  for (handle parent : reinterpret_borrow<tuple>(type->tp_bases))
144  check.push_back((PyTypeObject *) parent.ptr());
145  }
146  }
147 }

220 PYBIND11_NOINLINE handle find_registered_python_instance(void *src,
221  const detail::type_info *tinfo) {
222  auto it_instances = get_internals().registered_instances.equal_range(src);

233  instance *inst = nullptr;
234  size_t index = 0u;
235  const detail::type_info *type = nullptr;
236  void **vh = nullptr;

238  // Main constructor for a found value/holder:
239  value_and_holder(instance *i, const detail::type_info *type, size_t vpos, size_t index) :
240  inst{i}, index{index}, type{type},
241  vh{inst->simple_layout ? inst->simple_value_holder : &inst->nonsimple.values_and_holders[vpos]}
242  {}
243 
244  // Default constructor (used to signal a value-and-holder not found by get_value_and_holder())
245  value_and_holder() = default;
246 
247  // Used for past-the-end iterator
248  explicit value_and_holder(size_t index) : index{index} {}
249 
250  template <typename V = void> V *&value_ptr() const {
251  return reinterpret_cast<V *&>(vh[0]);
252  }
253  // True if this `value_and_holder` has a non-null value pointer
254  explicit operator bool() const { return value_ptr() != nullptr; }
255 
256  template <typename H> H &holder() const {
257  return reinterpret_cast<H &>(vh[1]);
258  }
259  bool holder_constructed() const {
260  return inst->simple_layout
261  ? inst->simple_holder_constructed
262  : (inst->nonsimple.status[index] & instance::status_holder_constructed) != 0u;
263  }
264  // NOLINTNEXTLINE(readability-make-member-function-const)
265  void set_holder_constructed(bool v = true) {
266  if (inst->simple_layout)
267  inst->simple_holder_constructed = v;
268  else if (v)
269  inst->nonsimple.status[index] |= instance::status_holder_constructed;
270  else
271  inst->nonsimple.status[index] &= (std::uint8_t) ~instance::status_holder_constructed;
272  }
273  bool instance_registered() const {
274  return inst->simple_layout
275  ? inst->simple_instance_registered
276  : ((inst->nonsimple.status[index] & instance::status_instance_registered) != 0);
277  }
278  // NOLINTNEXTLINE(readability-make-member-function-const)
279  void set_instance_registered(bool v = true) {
280  if (inst->simple_layout)
281  inst->simple_instance_registered = v;
282  else if (v)
283  inst->nonsimple.status[index] |= instance::status_instance_registered;
284  else
285  inst->nonsimple.status[index] &= (std::uint8_t) ~instance::status_instance_registered;
286  }

290 struct values_and_holders {
291 private:
292  instance *inst;
293  using type_vec = std::vector<detail::type_info *>;
294  const type_vec &tinfo;
295 
296 public:
297  explicit values_and_holders(instance *inst)
298  : inst{inst}, tinfo(all_type_info(Py_TYPE(inst))) {}

300  struct iterator {
301  private:
302  instance *inst = nullptr;
303  const type_vec *types = nullptr;
304  value_and_holder curr;
305  friend struct values_and_holders;
306  iterator(instance *inst, const type_vec *tinfo)
307  : inst{inst}, types{tinfo},
308  curr(inst /* instance */,
309  types->empty() ? nullptr : (*types)[0] /* type info */,
310  0, /* vpos: (non-simple types only): the first vptr comes first */
311  0 /* index */)
312  {}
313  // Past-the-end iterator:
314  explicit iterator(size_t end) : curr(end) {}
315 
316  public:
317  bool operator==(const iterator &other) const { return curr.index == other.curr.index; }
318  bool operator!=(const iterator &other) const { return curr.index != other.curr.index; }
319  iterator &operator++() {
320  if (!inst->simple_layout)
321  curr.vh += 1 + (*types)[curr.index]->holder_size_in_ptrs;
322  ++curr.index;
323  curr.type = curr.index < types->size() ? (*types)[curr.index] : nullptr;
324  return *this;
325  }
326  value_and_holder &operator*() { return curr; }
327  value_and_holder *operator->() { return &curr; }
328  };
329 
330  iterator begin() { return iterator(inst, &tinfo); }
331  iterator end() { return iterator(tinfo.size()); }
332 
333  iterator find(const type_info *find_type) {
334  auto it = begin(), endit = end();
335  while (it != endit && it->type != find_type) ++it;
336  return it;
337  }
338 
339  size_t size() { return tinfo.size(); }
340 };

353  // Optimize common case:

376 PYBIND11_NOINLINE void instance::allocate_layout() {
377  auto &tinfo = all_type_info(Py_TYPE(this));
378 
379  const size_t n_types = tinfo.size();
380 
381  if (n_types == 0)
382  pybind11_fail("instance allocation failed: new instance has no pybind11-registered base types");
383 
384  simple_layout =
385  n_types == 1 && tinfo.front()->holder_size_in_ptrs <= instance_simple_holder_in_ptrs();
386 
387  // Simple path: no python-side multiple inheritance, and a small-enough holder
388  if (simple_layout) {
389  simple_value_holder[0] = nullptr;
390  simple_holder_constructed = false;
391  simple_instance_registered = false;
392  }
393  else { // multiple base types or a too-large holder
394  // Allocate space to hold: [v1*][h1][v2*][h2]...[bb...] where [vN*] is a value pointer,
395  // [hN] is the (uninitialized) holder instance for value N, and [bb...] is a set of bool
396  // values that tracks whether each associated holder has been initialized. Each [block] is
397  // padded, if necessary, to an integer multiple of sizeof(void *).
398  size_t space = 0;
399  for (auto t : tinfo) {
400  space += 1; // value pointer
401  space += t->holder_size_in_ptrs; // holder instance
402  }
403  size_t flags_at = space;
404  space += size_in_ptrs(n_types); // status bytes (holder_constructed and instance_registered)
405 
406  // Allocate space for flags, values, and holders, and initialize it to 0 (flags and values,
407  // in particular, need to be 0). Use Python's memory allocation functions: in Python 3.6
408  // they default to using pymalloc, which is designed to be efficient for small allocations
409  // like the one we're doing here; in earlier versions (and for larger allocations) they are
410  // just wrappers around malloc.
411 #if PY_VERSION_HEX >= 0x03050000
412  nonsimple.values_and_holders = (void **) PyMem_Calloc(space, sizeof(void *));
413  if (!nonsimple.values_and_holders) throw std::bad_alloc();
414 #else
415  nonsimple.values_and_holders = (void **) PyMem_New(void *, space);
416  if (!nonsimple.values_and_holders) throw std::bad_alloc();
417  std::memset(nonsimple.values_and_holders, 0, space * sizeof(void *));
418 #endif
419  nonsimple.status = reinterpret_cast<std::uint8_t *>(&nonsimple.values_and_holders[flags_at]);
420  }
421  owned = true;
422 }

437 PYBIND11_NOINLINE std::string error_string() {
438  if (!PyErr_Occurred()) {
439  PyErr_SetString(PyExc_RuntimeError, "Unknown internal error occurred");
440  return "Unknown internal error occurred";
441  }
442 
443  error_scope scope; // Preserve error state
444 
445  std::string errorString;
446  if (scope.type) {
447  errorString += handle(scope.type).attr("__name__").cast<std::string>();
448  errorString += ": ";
449  }
450  if (scope.value)
451  errorString += (std::string) str(scope.value);
452 
453  PyErr_NormalizeException(&scope.type, &scope.value, &scope.trace);
454 
455 #if PY_MAJOR_VERSION >= 3
456  if (scope.trace != nullptr)
457  PyException_SetTraceback(scope.value, scope.trace);
458 #endif
459 
460 #if !defined(PYPY_VERSION)
461  if (scope.trace) {
462  auto *trace = (PyTracebackObject *) scope.trace;
463 
464  /* Get the deepest trace possible */
465  while (trace->tb_next)
466  trace = trace->tb_next;
467 
468  PyFrameObject *frame = trace->tb_frame;
469  errorString += "\n\nAt:\n";
470  while (frame) {
471 #if PY_VERSION_HEX >= 0x03090000
472  PyCodeObject *f_code = PyFrame_GetCode(frame);
473 #else
474  PyCodeObject *f_code = frame->f_code;
475  Py_INCREF(f_code);
476 #endif
477  int lineno = PyFrame_GetLineNumber(frame);
478  errorString +=
479  " " + handle(f_code->co_filename).cast<std::string>() +
480  "(" + std::to_string(lineno) + "): " +
481  handle(f_code->co_name).cast<std::string>() + "\n";
482  frame = frame->f_back;
483  Py_DECREF(f_code);
484  }
485  }
486 #endif
487 
488  return errorString;
489 }

533  PYBIND11_NOINLINE static handle cast(const void *_src, return_value_policy policy, handle parent,
534  const detail::type_info *tinfo,
535  void *(*copy_constructor)(const void *),
536  void *(*move_constructor)(const void *),
537  const void *existing_holder = nullptr) {
538  if (!tinfo) // no type info: error will be set already
539  return handle();
540 
541  void *src = const_cast<void *>(_src);
542  if (src == nullptr)
543  return none().release();
544 
545  if (handle registered_inst = find_registered_python_instance(src, tinfo))
546  return registered_inst;
547 
548  auto inst = reinterpret_steal<object>(make_new_instance(tinfo->type));
549  auto wrapper = reinterpret_cast<instance *>(inst.ptr());
550  wrapper->owned = false;
551  void *&valueptr = values_and_holders(wrapper).begin()->value_ptr();
552 
553  switch (policy) {
554  case return_value_policy::automatic:
555  case return_value_policy::take_ownership:
556  valueptr = src;
557  wrapper->owned = true;
558  break;
559 
560  case return_value_policy::automatic_reference:
561  case return_value_policy::reference:
562  valueptr = src;
563  wrapper->owned = false;
564  break;
565 
566  case return_value_policy::copy:
567  if (copy_constructor)
568  valueptr = copy_constructor(src);
569  else {
570 #if defined(NDEBUG)
571  throw cast_error("return_value_policy = copy, but type is "
572  "non-copyable! (compile in debug mode for details)");
573 #else
574  std::string type_name(tinfo->cpptype->name());
575  detail::clean_type_id(type_name);
576  throw cast_error("return_value_policy = copy, but type " +
577  type_name + " is non-copyable!");
578 #endif
579  }
580  wrapper->owned = true;
581  break;
582 
583  case return_value_policy::move:
584  if (move_constructor)
585  valueptr = move_constructor(src);
586  else if (copy_constructor)
587  valueptr = copy_constructor(src);
588  else {
589 #if defined(NDEBUG)
590  throw cast_error("return_value_policy = move, but type is neither "
591  "movable nor copyable! "
592  "(compile in debug mode for details)");
593 #else
594  std::string type_name(tinfo->cpptype->name());
595  detail::clean_type_id(type_name);
596  throw cast_error("return_value_policy = move, but type " +
597  type_name + " is neither movable nor copyable!");
598 #endif
599  }
600  wrapper->owned = true;
601  break;
602 
603  case return_value_policy::reference_internal:
604  valueptr = src;
605  wrapper->owned = false;
606  keep_alive_impl(inst, parent);
607  break;
608 
609  default:
610  throw cast_error("unhandled return_value_policy: should not happen!");
611  }
612 
613  tinfo->init_instance(wrapper, existing_holder);
614 
615  return inst.release();
616  }

618  // Base methods for generic caster; there are overridden in copyable_holder_caster
619  void load_value(value_and_holder &&v_h) {
620  auto *&vptr = v_h.value_ptr();
621  // Lazy allocation for unallocated values:
622  if (vptr == nullptr) {
623  auto *type = v_h.type ? v_h.type : typeinfo;
624  if (type->operator_new) {
625  vptr = type->operator_new(type->type_size);
626  } else {
627  #if defined(__cpp_aligned_new) && (!defined(_MSC_VER) || _MSC_VER >= 1912)
628  if (type->type_align > __STDCPP_DEFAULT_NEW_ALIGNMENT__)
629  vptr = ::operator new(type->type_size,
630  std::align_val_t(type->type_align));
631  else

685  // Implementation of `load`; this takes the type of `this` so that it can dispatch the relevant
686  // bits of code between here and copyable_holder_caster where the two classes need different
687  // logic (without having to resort to virtual inheritance).
688  template <typename ThisT>
689  PYBIND11_NOINLINE bool load_impl(handle src, bool convert) {
690  if (!src) return false;
691  if (!typeinfo) return try_load_foreign_module_local(src);
692 
693  auto &this_ = static_cast<ThisT &>(*this);
694  this_.check_holder_compat();
695 
696  PyTypeObject *srctype = Py_TYPE(src.ptr());
697 
698  // Case 1: If src is an exact type match for the target type then we can reinterpret_cast
699  // the instance's value pointer to the target type:
700  if (srctype == typeinfo->type) {
701  this_.load_value(reinterpret_cast<instance *>(src.ptr())->get_value_and_holder());
702  return true;
703  }
704  // Case 2: We have a derived class
705  if (PyType_IsSubtype(srctype, typeinfo->type)) {
706  auto &bases = all_type_info(srctype);
707  bool no_cpp_mi = typeinfo->simple_type;
708 
709  // Case 2a: the python type is a Python-inherited derived class that inherits from just
710  // one simple (no MI) pybind11 class, or is an exact match, so the C++ instance is of
711  // the right type and we can use reinterpret_cast.
712  // (This is essentially the same as case 2b, but because not using multiple inheritance
713  // is extremely common, we handle it specially to avoid the loop iterator and type
714  // pointer lookup overhead)
715  if (bases.size() == 1 && (no_cpp_mi || bases.front()->type == typeinfo->type)) {
716  this_.load_value(reinterpret_cast<instance *>(src.ptr())->get_value_and_holder());
717  return true;
718  }
719  // Case 2b: the python type inherits from multiple C++ bases. Check the bases to see if
720  // we can find an exact match (or, for a simple C++ type, an inherited match); if so, we
721  // can safely reinterpret_cast to the relevant pointer.
722  if (bases.size() > 1) {
723  for (auto base : bases) {
724  if (no_cpp_mi ? PyType_IsSubtype(base->type, typeinfo->type) : base->type == typeinfo->type) {
725  this_.load_value(reinterpret_cast<instance *>(src.ptr())->get_value_and_holder(base));
726  return true;
727  }
728  }
729  }
730 
731  // Case 2c: C++ multiple inheritance is involved and we couldn't find an exact type match
732  // in the registered bases, above, so try implicit casting (needed for proper C++ casting
733  // when MI is involved).
734  if (this_.try_implicit_casts(src, convert))
735  return true;
736  }
737 
738  // Perform an implicit conversion
739  if (convert) {
740  for (auto &converter : typeinfo->implicit_conversions) {
741  auto temp = reinterpret_steal<object>(converter(src.ptr(), typeinfo->type));
742  if (load_impl<ThisT>(temp, false)) {
743  loader_life_support::add_patient(temp);
744  return true;
745  }
746  }
747  if (this_.try_direct_conversions(src))
748  return true;
749  }
750 
751  // Failed to match local typeinfo. Try again with global.
752  if (typeinfo->module_local) {
753  if (auto gtype = get_global_type_info(*typeinfo->cpptype)) {
754  typeinfo = gtype;
755  return load(src, false);
756  }
757  }
758 
759  // Global typeinfo has precedence over foreign module_local
760  if (try_load_foreign_module_local(src)) {
761  return true;
762  }
763 
764  // Custom converters didn't take None, now we convert None to nullptr.
765  if (src.is_none()) {
766  // Defer accepting None to other overloads (if we aren't in convert mode):
767  if (!convert) return false;
768  value = nullptr;
769  return true;
770  }
771 
772  return false;
773  }

856 // polymorphic_type_hook<itype>::get(src, tinfo) determines whether the object pointed
857 // to by `src` actually is an instance of some class derived from `itype`.
858 // If so, it sets `tinfo` to point to the std::type_info representing that derived
859 // type, and returns a pointer to the start of the most-derived object of that type
860 // (in which `src` is a subobject; this will be the same address as `src` in most
861 // single inheritance cases). If not, or if `src` is nullptr, it simply returns `src`
862 // and leaves `tinfo` at its default value of nullptr.
863 //
864 // The default polymorphic_type_hook just returns src. A specialization for polymorphic
865 // types determines the runtime type of the passed object and adjusts the this-pointer
866 // appropriately via dynamic_cast<void*>. This is what enables a C++ Animal* to appear
867 // to Python as a Dog (if Dog inherits from Animal, Animal is polymorphic, Dog is
868 // registered with pybind11, and this Animal is in fact a Dog).
869 //
870 // You may specialize polymorphic_type_hook yourself for types that want to appear
871 // polymorphic to Python but do not use C++ RTTI. (This is a not uncommon pattern
872 // in performance-sensitive applications, used most notably in LLVM.)
873 //
874 // polymorphic_type_hook_base allows users to specialize polymorphic_type_hook with
875 // std::enable_if. User provided specializations will always have higher priority than
876 // the default implementation and specialization provided in polymorphic_type_hook_base.
877 template <typename itype, typename SFINAE = void>
878 struct polymorphic_type_hook_base
879 {
880  static const void *get(const itype *src, const std::type_info*&) { return src; }
881 };
882 template <typename itype>
883 struct polymorphic_type_hook_base<itype, detail::enable_if_t<std::is_polymorphic<itype>::value>>
884 {
885  static const void *get(const itype *src, const std::type_info*& type) {
886  type = src ? &typeid(*src) : nullptr;
887  return dynamic_cast<const void*>(src);
888  }
889 };
