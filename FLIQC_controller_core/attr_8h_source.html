313  PYBIND11_NOINLINE void add_base(const std::type_info &base, void *(*caster)(void *)) {
314  auto base_info = detail::get_type_info(base, false);
315  if (!base_info) {
316  std::string tname(base.name());
317  detail::clean_type_id(tname);
318  pybind11_fail("generic_type: type \"" + std::string(name) +
319  "\" referenced unknown base type \"" + tname + "\"");
320  }
321 
322  if (default_holder != base_info->default_holder) {
323  std::string tname(base.name());
324  detail::clean_type_id(tname);
325  pybind11_fail("generic_type: type \"" + std::string(name) + "\" " +
326  (default_holder ? "does not have" : "has") +
327  " a non-default holder type while its base \"" + tname + "\" " +
328  (base_info->default_holder ? "does not" : "does"));
329  }
330 
331  bases.append((PyObject *) base_info->type);
332 
333  if (base_info->type->tp_dictoffset != 0)
334  dynamic_attr = true;
335 
336  if (caster)
337  base_info->implicit_casts.emplace_back(type, caster);
338  }

434  static void init(const arg_v &a, function_record *r) {
435  if (r->is_method && r->args.empty())
436  r->args.emplace_back("self", /*descr=*/ nullptr, /*parent=*/ handle(), /*convert=*/ true, /*none=*/ false);
437 
438  if (!a.value) {
439 #if !defined(NDEBUG)
440  std::string descr("'");

442  descr += a.type + "'";

444  if (r->name)

446  else
447  descr += " in method of '" + (std::string) str(r->scope) + "'";
448  } else if (r->name) {
449  descr += " in function '" + (std::string) r->name + "'";
450  }
451  pybind11_fail("arg(): could not convert default argument "
452  + descr + " into a Python object (type not registered yet?)");
453 #else

455  "into a Python object (type not registered yet?). "
456  "Compile in debug mode for more information.");
